Introdução ao React

React é uma biblioteca JavaScript para construção de interfaces de usuário. Desenvolvida pelo Facebook, permite criar componentes reutilizáveis e gerenciar o estado da aplicação de forma eficiente.

// Exemplo de um componente React básico
import React from 'react';

function Saudacao() {
    return (
        

            
Olá, Mundo!

            
Bem-vindo ao React

        

    );
}

export default Saudacao;

JSX

JSX é uma extensão de sintaxe para JavaScript que permite escrever código HTML dentro do JavaScript. O React usa JSX para descrever como a interface deve parecer.

// Exemplo de JSX
const elemento = (
    

        
Olá, {nome}

        
Bem-vindo ao nosso site!

    

);

Componentes e Props

Componentes são blocos de construção reutilizáveis em React. Props são propriedades que podem ser passadas para componentes.

// Componente com props
function Saudacao(props) {
    return (
        

            
Olá, {props.nome}!

            
Você tem {props.idade} anos.

        

    );
}

// Uso do componente

Conceitos Intermediários

Estado (State)

O estado é um objeto que armazena dados que podem mudar ao longo do tempo em um componente.

// Exemplo de uso de estado
import React, { useState } from 'react';

function Contador() {
    const [contador, setContador] = useState(0);

    return (
        

            
Contador: {contador}

            
        

    );
}

Efeitos (useEffect)

O useEffect é um Hook que permite executar efeitos colaterais em componentes funcionais.

// Exemplo de useEffect
import React, { useState, useEffect } from 'react';

function Exemplo() {
    const [dados, setDados] = useState([]);

    useEffect(() => {
        // Buscar dados da API
        fetch('https://api.exemplo.com/dados')
            .then(response => response.json())
            .then(data => setDados(data));
    }, []); // Array de dependências vazio = executa apenas uma vez

    return (
        

            {dados.map(item => (
                
{item.nome}

            ))}
        

    );
}

Conceitos Avançados

Context API

A Context API permite compartilhar dados entre componentes sem precisar passar props manualmente em cada nível.

// Criando um contexto
import React, { createContext, useContext, useState } from 'react';

const TemaContext = createContext();

function ProvedorTema({ children }) {
    const [tema, setTema] = useState('claro');

    return (
        
            {children}
        
    );
}

// Usando o contexto
function Componente() {
    const { tema, setTema } = useContext(TemaContext);
    return (
        

            
Tema atual: {tema}

            
        

    );
}

Hooks Personalizados

Hooks personalizados permitem extrair lógica de componentes para funções reutilizáveis.

// Exemplo de hook personalizado
function useContador(valorInicial = 0) {
    const [contador, setContador] = useState(valorInicial);

    const incrementar = () => setContador(c => c + 1);
    const decrementar = () => setContador(c => c - 1);
    const resetar = () => setContador(valorInicial);

    return { contador, incrementar, decrementar, resetar };
}

// Usando o hook personalizado
function Componente() {
    const { contador, incrementar, decrementar } = useContador(0);

    return (
        

            
Contador: {contador}

            
            
        

    );
}

Renderização Condicional

React permite renderizar componentes de forma condicional usando operadores lógicos e ternários.

// Exemplo de renderização condicional
function Componente({ usuario }) {
    return (
        

            {usuario ? (
                

                    
Bem-vindo, {usuario.nome}!

                    
                

            ) : (
                

                    
Por favor, faça login

                    
                

            )}
        

    );
}
