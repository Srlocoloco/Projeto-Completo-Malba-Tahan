HTML Básico

O que é HTML?

HTML (HyperText Markup Language) é a linguagem fundamental para criar páginas web. É uma linguagem de marcação que usa tags para estruturar o conteúdo.

  • HyperText: Texto que contém links para outras páginas
  • Markup: Sistema de marcação usando tags
  • Language: Linguagem com regras e sintaxe próprias

Estrutura Básica de um Documento HTML

<!DOCTYPE html>  
<html lang="pt-BR">  
<head>  
    <meta charset="UTF-8">  
    <meta name="viewport" content="width=device-width, initial-scale=1.0">  
    <title>Minha Página</title>  
</head>
<body>  
    <!-- Conteúdo da página aqui -->
</body>
</html>

HTML Avançado

Elementos Semânticos

O HTML5 introduziu vários elementos semânticos que ajudam a estruturar melhor o conteúdo:

  • <header> - Cabeçalho da página ou seção
  • <nav> - Menu de navegação
  • <main> - Conteúdo principal
  • <article> - Conteúdo independente e autocontido
  • <section> - Seção temática de conteúdo
  • <aside> - Conteúdo relacionado (sidebar)
  • <footer> - Rodapé da página ou seção
  • <figure> - Conteúdo independente (imagens, diagramas)
  • <figcaption> - Legenda para figure
  • <time> - Representa data/hora
  • <mark> - Texto marcado/destacado
  • <details> e <summary> - Conteúdo expansível
  • <dialog> - Caixa de diálogo
  • <data> - Dados com valor semântico
  • <meter> - Medida dentro de um intervalo
  • <progress> - Barra de progresso

Atributos Globais

Atributos que podem ser usados em qualquer elemento HTML:

  • class - Define classes CSS
  • id - Identificador único
  • style - Estilos CSS inline
  • title - Informação adicional (tooltip)
  • lang - Idioma do elemento
  • data-* - Atributos personalizados
  • hidden - Esconde o elemento
  • tabindex - Ordem de foco
  • contenteditable - Permite edição
  • spellcheck - Verificação ortográfica
  • draggable - Permite arrastar o elemento
  • dropzone - Define onde elementos podem ser soltos
  • translate - Controla se o texto deve ser traduzido
  • dir - Direção do texto (ltr/rtl)
  • accesskey - Atalho de teclado

<script>
    // Funções de validação e interatividade
    function validarUsuario(input) {
        const feedback = input.nextElementSibling;
        const valor = input.value;
        
        if (valor.length < 3) {
            feedback.textContent = 'Mínimo de 3 caracteres';
            feedback.style.color = 'red';
        } else if (valor.length > 20) {
            feedback.textContent = 'Máximo de 20 caracteres';
            feedback.style.color = 'red';
        } else if (!/^[a-zA-Z0-9]+$/.test(valor)) {
            feedback.textContent = 'Apenas letras e números';
            feedback.style.color = 'red';
        } else {
            feedback.textContent = '✓ Válido';
            feedback.style.color = 'green';
        }
    }

    function previewImagem(input) {
        const preview = document.getElementById('preview');
        const file = input.files[0];
        
        if (file) {
            const reader = new FileReader();
            reader.onload = function(e) {
                preview.innerHTML = `<img src="${e.target.result}" style="max-width: 200px;">`;
            }
            reader.readAsDataURL(file);
        }
    }

    function atualizarTema(cor) {
        const preview = document.getElementById('previewCor');
        preview.style.backgroundColor = cor;
        preview.style.width = '50px';
        preview.style.height = '50px';
        preview.style.borderRadius = '50%';
    }

    function validarSenha(input) {
        const barra = document.querySelector('.senha-forca .barra');
        const texto = document.querySelector('.senha-forca .texto');
        const senha = input.value;
        
        let forca = 0;
        if (senha.length >= 8) forca++;
        if (/[A-Z]/.test(senha)) forca++;
        if (/[a-z]/.test(senha)) forca++;
        if (/[0-9]/.test(senha)) forca++;
        if (/[^A-Za-z0-9]/.test(senha)) forca++;
        
        const cores = ['red', 'orange', 'yellow', 'lightgreen', 'green'];
        const textos = ['Muito fraca', 'Fraca', 'Média', 'Forte', 'Muito forte'];
        
        barra.style.width = `${forca * 20}%`;
        barra.style.backgroundColor = cores[forca - 1];
        texto.textContent = textos[forca - 1];
    }

    function mostrarCamposAdicionais(tipo) {
        const campos = document.getElementById('camposEmpresariais');
        campos.style.display = tipo === 'empresarial' ? 'block' : 'none';
    }

    function validarFormulario() {
        const form = document.getElementById('formularioAvancado');
        if (!form.checkValidity()) {
            alert('Por favor, preencha todos os campos obrigatórios corretamente.');
            return false;
        }
        return true;
    }
</script>

<script>
    // Funções de acessibilidade
    function togglePainel(button) {
        const painel = document.getElementById('painelNotificacoes');
        const expanded = button.getAttribute('aria-expanded') === 'true';
        
        button.setAttribute('aria-expanded', !expanded);
        painel.hidden = expanded;
    }

    function fecharModal() {
        const modal = document.querySelector('[role="dialog"]');
        modal.style.display = 'none';
    }

    // Gerenciamento de tabs
    document.querySelectorAll('[role="tab"]').forEach(tab => {
        tab.addEventListener('click', () => {
            // Desativa todas as tabs
            document.querySelectorAll('[role="tab"]').forEach(t => {
                t.setAttribute('aria-selected', 'false');
            });
            
            // Esconde todos os painéis
            document.querySelectorAll('[role="tabpanel"]').forEach(p => {
                p.hidden = true;
            });
            
            // Ativa a tab clicada
            tab.setAttribute('aria-selected', 'true');
            
            // Mostra o painel correspondente
            const painelId = tab.getAttribute('aria-controls');
            document.getElementById(painelId).hidden = false;
        });
    });
</script>
